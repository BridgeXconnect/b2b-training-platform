**Source: components/learning/AIChatInterface.tsx**
- Existing CopilotKit integration patterns and action structure
- AI service communication and streaming response handling

**Advanced Action Models Required:**
```typescript
export interface AdvancedCopilotAction {
  id: string;
  name: string;
  description: string;
  parameters: Record<string, any>;
  execute: (params: any, context: UserContext) => Promise<ActionResult>;
  validate: (params: any) => ValidationResult;
  fallback?: (error: Error, params: any) => Promise<ActionResult>;
  monitoring: {
    executionTime: number;
    successRate: number;
    errorRate: number;
  };
}

export interface WorkflowDefinition {
  id: string;
  name: string;
  description: string;
  steps: WorkflowStep[];
  triggers: WorkflowTrigger[];
  conditions: WorkflowCondition[];
  state: WorkflowState;
  metadata: {
    created: Date;
    updated: Date;
    version: string;
    author: string;
  };
}

export interface RecommendationEngine {
  generateRecommendations(
    userId: string,
    context: LearningContext,
    preferences: UserPreferences
  ): Promise<Recommendation[]>;
  
  updateUserModel(
    userId: string,
    interactions: UserInteraction[]
  ): Promise<void>;
  
  getRecommendationExplanation(
    recommendationId: string
  ): Promise<RecommendationExplanation>;
}
```
